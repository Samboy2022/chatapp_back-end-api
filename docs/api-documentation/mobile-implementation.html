Quick Start

Get started with React Native integration in 5 minutes:

1. Install Dependencies
npm install @react-native-async-storage/async-storage
npm install laravel-echo pusher-js
npm install @react-native-community/netinfo
2. Configure API Service
// services/ApiService.js
import AsyncStorage from '@react-native-async-storage/async-storage';

class ApiService {
  constructor() {
    this.baseURL = 'https://your-api-url.com/api';
    this.token = null;
  }

  async initialize() {
    this.token = await AsyncStorage.getItem('auth_token');
  }

  async request(endpoint, options = {}) {
    const url = `${this.baseURL}${endpoint}`;
    const config = {
      headers: {
        'Content-Type': 'application/json',
        'Accept': 'application/json',
        ...(this.token && { 'Authorization': `Bearer ${this.token}` }),
      },
      ...options,
    };

    const response = await fetch(url, config);
    const data = await response.json();

    if (!response.ok) {
      throw new Error(data.message || 'API request failed');
    }

    return data;
  }

  // Authentication methods
  async login(credentials) {
    const data = await this.request('/auth/login', {
      method: 'POST',
      body: JSON.stringify(credentials),
    });

    if (data.success) {
      this.token = data.data.token;
      await AsyncStorage.setItem('auth_token', this.token);
      await AsyncStorage.setItem('user_data', JSON.stringify(data.data.user));
    }

    return data;
  }

  async logout() {
    try {
      await this.request('/auth/logout', { method: 'POST' });
    } finally {
      this.token = null;
      await AsyncStorage.multiRemove(['auth_token', 'user_data']);
    }
  }

  // Chat methods
  async getChats() {
    return this.request('/chats');
  }

  async sendMessage(chatId, message) {
    return this.request(`/chats/${chatId}/messages`, {
      method: 'POST',
      body: JSON.stringify(message),
    });
  }

  // Call methods
  async initiateCall(receiverId, callType) {
    return this.request('/calls', {
      method: 'POST',
      body: JSON.stringify({
        receiver_id: receiverId,
        type: callType,
      }),
    });
  }

  async answerCall(callId) {
    return this.request(`/calls/${callId}/answer`, {
      method: 'POST',
    });
  }

  async endCall(callId) {
    return this.request(`/calls/${callId}/end`, {
      method: 'POST',
    });
  }
}

export default new ApiService();
3. Set up WebSocket Service
// services/WebSocketService.js
import Echo from 'laravel-echo';
import Pusher from 'pusher-js';
import ApiService from './ApiService';

class WebSocketService {
  constructor() {
    this.echo = null;
    this.isConnected = false;
  }

  async initialize(userId) {
    try {
      // Get broadcast configuration
      const config = await this.getBroadcastConfig();
      
      if (!config.enabled) {
        throw new Error('Broadcasting is disabled');
      }

      // Initialize Echo based on driver
      if (config.driver === 'pusher') {
        this.initializePusher(config.config);
      } else if (config.driver === 'reverb') {
        this.initializeReverb(config.config);
      }

      this.isConnected = true;
      return true;
    } catch (error) {
      console.error('Failed to initialize WebSocket:', error);
      return false;
    }
  }

  async getBroadcastConfig() {
    const response = await fetch('https://your-api-url.com/api/broadcast-settings');
    const data = await response.json();
    return data.success ? data.data : null;
  }

  initializePusher(config) {
    window.Pusher = Pusher;
    
    this.echo = new Echo({
      broadcaster: 'pusher',
      key: config.key,
      cluster: config.cluster,
      forceTLS: true,
      auth: {
        headers: {
          Authorization: `Bearer ${ApiService.token}`,
        },
      },
      authEndpoint: config.auth_endpoint,
    });
  }

  initializeReverb(config) {
    this.echo = new Echo({
      broadcaster: 'socket.io',
      host: `${config.scheme}://${config.host}:${config.port}`,
      auth: {
        headers: {
          Authorization: `Bearer ${ApiService.token}`,
        },
      },
      authEndpoint: config.auth_endpoint,
    });
  }

  subscribeToChannel(channelName, events) {
    if (!this.echo) return null;

    const channel = this.echo.private(channelName);
    
    Object.entries(events).forEach(([eventName, callback]) => {
      channel.listen(eventName, callback);
    });

    return channel;
  }

  disconnect() {
    if (this.echo) {
      this.echo.disconnect();
      this.echo = null;
      this.isConnected = false;
    }
  }
}

export default new WebSocketService();

Authentication Provider

Create an authentication provider to manage user state:

// providers/auth_provider.dart
import 'dart:convert';
import 'package:flutter/foundation.dart';
import 'package:shared_preferences/shared_preferences.dart';
import '../services/api_service.dart';
import '../services/websocket_service.dart';

class User {
  final int id;
  final String name;
  final String email;
  final String? avatarUrl;

  User({
    required this.id,
    required this.name,
    required this.email,
    this.avatarUrl,
  });

  factory User.fromJson(Map<String, dynamic> json) {
    return User(
      id: json['id'],
      name: json['name'],
      email: json['email'],
      avatarUrl: json['avatar_url'],
    );
  }
}

class AuthProvider with ChangeNotifier {
  final ApiService _apiService = ApiService();
  final WebSocketService _webSocketService = WebSocketService();

  User? _user;
  bool _isLoading = false;
  bool _isInitialized = false;

  User? get user => _user;
  bool get isLoading => _isLoading;
  bool get isAuthenticated => _user != null;
  bool get isInitialized => _isInitialized;

  Future<void> initialize() async {
    if (_isInitialized) return;

    _isLoading = true;
    notifyListeners();

    try {
      await _apiService.initialize();

      if (_apiService.token != null) {
        final prefs = await SharedPreferences.getInstance();
        final userData = prefs.getString('user_data');

        if (userData != null) {
          _user = User.fromJson(json.decode(userData));

          // Initialize WebSocket connection
          await _webSocketService.initialize(
            userId: _user!.id.toString(),
            authToken: _apiService.token!,
          );
        }
      }
    } catch (error) {
      debugPrint('Failed to initialize auth: $error');
    } finally {
      _isLoading = false;
      _isInitialized = true;
      notifyListeners();
    }
  }

  Future<bool> login(String email, String password) async {
    _isLoading = true;
    notifyListeners();

    try {
      final response = await _apiService.login(email, password);

      if (response['success'] == true) {
        _user = User.fromJson(response['data']['user']);

        // Initialize WebSocket connection
        await _webSocketService.initialize(
          userId: _user!.id.toString(),
          authToken: _apiService.token!,
        );

        notifyListeners();
        return true;
      }

      return false;
    } catch (error) {
      debugPrint('Login failed: $error');
      return false;
    } finally {
      _isLoading = false;
      notifyListeners();
    }
  }

  Future<void> logout() async {
    _isLoading = true;
    notifyListeners();

    try {
      // Disconnect WebSocket
      await _webSocketService.dispose();

      // Logout from API
      await _apiService.logout();

      // Clear user state
      _user = null;
    } catch (error) {
      debugPrint('Logout failed: $error');
    } finally {
      _isLoading = false;
      notifyListeners();
    }
  }
}
Main App Setup
// main.dart
import 'package:flutter/material.dart';
import 'package:provider/provider.dart';
import 'providers/auth_provider.dart';
import 'screens/splash_screen.dart';
import 'screens/login_screen.dart';
import 'screens/home_screen.dart';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return ChangeNotifierProvider(
      create: (context) => AuthProvider(),
      child: MaterialApp(
        title: 'FarmersNetwork Chat',
        theme: ThemeData(
          primarySwatch: Colors.blue,
        ),
        home: AuthWrapper(),
      ),
    );
  }
}

class AuthWrapper extends StatefulWidget {
  @override
  _AuthWrapperState createState() => _AuthWrapperState();
}

class _AuthWrapperState extends State<AuthWrapper> {
  @override
  void initState() {
    super.initState();
    // Initialize authentication on app start
    WidgetsBinding.instance.addPostFrameCallback((_) {
      context.read<AuthProvider>().initialize();
    });
  }

  @override
  Widget build(BuildContext context) {
    return Consumer<AuthProvider>(
      builder: (context, authProvider, child) {
        if (!authProvider.isInitialized) {
          return SplashScreen();
        }

        if (authProvider.isAuthenticated) {
          return HomeScreen();
        } else {
          return LoginScreen();
        }
      },
    );
  }
}

WebSocket Implementation

Real-time WebSocket integration for both React Native and Flutter applications.

WebSocket Architecture

The FarmersNetwork Chat API supports two WebSocket drivers:

Pusher Cloud
  • Managed WebSocket service
  • Reliable and scalable
  • Easy to implement
  • Requires Pusher account
Laravel Reverb
  • Self-hosted WebSocket server
  • No external dependencies
  • Complete control
  • Requires server setup

Dynamic Configuration

The mobile app should dynamically detect and use the configured WebSocket driver:

// React Native example
async function getBroadcastConfig() {
  try {
    const response = await fetch('https://your-api-url.com/api/broadcast-settings');
    const data = await response.json();

    if (data.success) {
      return data.data;
    }

    return null;
  } catch (error) {
    console.error('Failed to get broadcast config:', error);
    return null;
  }
}

async function initializeWebSocket(authToken) {
  const config = await getBroadcastConfig();

  if (!config || !config.enabled) {
    console.warn('Broadcasting is disabled');
    return false;
  }

  if (config.driver === 'pusher') {
    // Initialize Pusher
    return initializePusher(config.config, authToken);
  } else if (config.driver === 'reverb') {
    // Initialize Reverb
    return initializeReverb(config.config, authToken);
  }

  return false;
}
// Flutter example
Future<Map<String, dynamic>?> getBroadcastConfig() async {
  try {
    final response = await http.get(
      Uri.parse('https://your-api-url.com/api/broadcast-settings'),
      headers: {
        'Accept': 'application/json',
      },
    );

    if (response.statusCode == 200) {
      final data = json.decode(response.body);
      if (data['success'] == true) {
        return data['data'];
      }
    }

    return null;
  } catch (error) {
    debugPrint('Failed to get broadcast config: $error');
    return null;
  }
}

Future<bool> initializeWebSocket(String authToken) async {
  final config = await getBroadcastConfig();

  if (config == null || config['enabled'] != true) {
    debugPrint('Broadcasting is disabled');
    return false;
  }

  if (config['driver'] == 'pusher') {
    // Initialize Pusher
    return initializePusher(config['config'], authToken);
  } else if (config['driver'] == 'reverb') {
    // Initialize Reverb
    return initializeReverb(config['config'], authToken);
  }

  return false;
}

Environment Configuration

Configure different environments for development, staging, and production:

React Native (.env files)
# .env.development
API_URL=http://192.168.0.102:8000/api
WEBSOCKET_HOST=http://192.168.0.102:6001

# .env.production
API_URL=https://api.farmersnetwork.com/api
WEBSOCKET_HOST=https://socket.farmersnetwork.com
Flutter (environment config)
// config/environment.dart
enum Environment { development, staging, production }

class EnvironmentConfig {
  static Environment _environment = Environment.development;

  static void setEnvironment(Environment environment) {
    _environment = environment;
  }

  static String get apiUrl {
    switch (_environment) {
      case Environment.development:
        return 'http://192.168.0.102:8000/api';
      case Environment.staging:
        return 'https://staging-api.farmersnetwork.com/api';
      case Environment.production:
        return 'https://api.farmersnetwork.com/api';
    }
  }

  static String get websocketHost {
    switch (_environment) {
      case Environment.development:
        return 'http://192.168.0.102:6001';
      case Environment.staging:
        return 'https://staging-socket.farmersnetwork.com';
      case Environment.production:
        return 'https://socket.farmersnetwork.com';
    }
  }
}
