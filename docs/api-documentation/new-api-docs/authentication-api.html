POST /auth/register

Register a new user account

Request Body

{
  "name": "John Doe",
  "email": "john@example.com",
  "phone_number": "+1234567890",
  "country_code": "+1",
  "password": "password123",
  "password_confirmation": "password123"
}

Success Response (201)

{
  "success": true,
  "message": "User registered successfully",
  "data": {
    "user": {
      "id": 1,
      "name": "John Doe",
      "email": "john@example.com",
      "phone_number": "+1234567890",
      "country_code": "+1"
    },
    "token": "1|abc123def456",
    "token_type": "Bearer"
  }
}

Required Fields

  • ‚Ä¢ name: User's full name
  • ‚Ä¢ email: Valid email address (unique)
  • ‚Ä¢ phone_number: Phone with country code
  • ‚Ä¢ country_code: Country code (e.g., +1)
  • ‚Ä¢ password: Minimum 8 characters
  • ‚Ä¢ password_confirmation: Must match password

Authenticate user and receive access token

Request Body

{
  "login": "+1234567890",
  "password": "password123"
}

Use phone number or email for login field

Success Response (200)

{
  "success": true,
  "message": "Login successful",
  "data": {
    "user": {
      "id": 1,
      "name": "John Doe",
      "email": "john@example.com",
      "phone_number": "+1234567890",
      "country_code": "+1",
      "is_online": true,
      "last_seen_at": "2025-09-15T00:36:00.000000Z"

PUT /auth/profile

{

  "about": "Updated bio",
  "email": "newemail@example.com",
  "current_password": "oldpassword123",
  "new_password": "newpassword123",
  "new_password_confirmation": "newpassword123"
}

Success Response (200)

{
  "success": true,
  "data": {
    "id": 1,
    "name": "Updated Name",
    "email": "newemail@example.com",
    "about": "Updated bio"
  },
  "message": "Profile updated successfully"
}
PUT /auth/privacy

Update user's privacy settings

Privacy Options

last_seen_privacy: everyone, contacts, nobody
profile_photo_privacy: everyone, contacts, nobody
about_privacy: everyone, contacts, nobody
status_privacy: everyone, contacts, close_friends
read_receipts_enabled: true, false
groups_privacy: everyone, contacts

Request Example

{
  "last_seen_privacy": "contacts",
  "read_receipts_enabled": true,
  "profile_photo_privacy": "everyone"
}

üíª Code Examples

JavaScript (Fetch API)

// Register new user
const registerUser = async (userData) => {
  const response = await fetch('http://127.0.0.1:8000/api/auth/register', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(userData)
  });
  return await response.json();
};

// Login user
const loginUser = async (login, password) => {
  const response = await fetch('http://127.0.0.1:8000/api/auth/login', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ login, password })
  });
  const data = await response.json();
  if (data.success) {
    localStorage.setItem('token', data.data.token);
  }
  return data;
};

// Get user profile
const getUserProfile = async () => {
  const token = localStorage.getItem('token');
  const response = await fetch('http://127.0.0.1:8000/api/auth/user', {
    headers: { 'Authorization': `Bearer ${token}` }
  });
  return await response.json();
};

cURL Commands

# Register user
curl -X POST http://127.0.0.1:8000/api/auth/register \
  -H "Content-Type: application/json" \
  -d '{
    "name": "John Doe",
    "email": "john@example.com",
    "phone_number": "+1234567890",
    "country_code": "+1",
    "password": "password123",
    "password_confirmation": "password123"
  }'

# Login user
curl -X POST http://127.0.0.1:8000/api/auth/login \
  -H "Content-Type: application/json" \
  -d '{
    "login": "+1234567890",
    "password": "password123"
  }'

# Get user profile
curl -X GET http://127.0.0.1:8000/api/auth/user \
  -H "Authorization: Bearer YOUR_TOKEN"

# Update profile
curl -X PUT http://127.0.0.1:8000/api/auth/profile \
  -H "Authorization: Bearer YOUR_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "name": "Updated Name",
    "about": "Updated bio"
  }'
