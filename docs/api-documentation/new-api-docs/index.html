// lib/repositories/base_repository.dart
abstract class BaseRepository {
  final String baseUrl = 'https://yourdomain.com/api';
  final Dio _dio = Dio();

  BaseRepository() {
    _setupDio();
  }

  void _setupDio() {
    _dio.options.baseUrl = baseUrl;
    _dio.options.connectTimeout = Duration(seconds: 30);
    _dio.options.receiveTimeout = Duration(seconds: 30);
    _dio.options.headers = {
      'Content-Type': 'application/json',
      'Accept': 'application/json',
    };

    // Add interceptors for token management
    _dio.interceptors.add(InterceptorsWrapper(
      onRequest: (options, handler) async {
        final token = await _getToken();
        if (token != null) {
          options.headers['Authorization'] = 'Bearer $token';
        }
        handler.next(options);
      },
      onError: (error, handler) async {
        if (error.response?.statusCode == 401) {
          await _refreshToken();
          handler.resolve(await _dio.fetch(error.requestOptions));
        } else {
          handler.next(error);
        }
      },
    ));
  }

  Future _getToken() async {
    final storage = FlutterSecureStorage();
    return await storage.read(key: 'auth_token');
  }

  Future _refreshToken() async {
    // Implement token refresh logic
  }
}

// lib/repositories/auth_repository.dart
class AuthRepository extends BaseRepository {
  // Login
  Future> login(String login, String password) async {
    try {
      final response = await _dio.post('/auth/login', data: {
        'login': login,    // email or phone
        'password': password,
      });

      if (response.statusCode == 200) {
        final data = response.data;
        if (data['success'] == true) {
          // Store token
          final token = data['data']['token'];
          final storage = FlutterSecureStorage();
          await storage.write(key: 'auth_token', value: token);

          return ApiResponse.success(LoginResponse.fromJson(data['data']));
        }
      }

      return ApiResponse.error('Login failed');
    } catch (e) {
      return ApiResponse.error('Network error: $e');
    }
  }

  // Get User Profile
  Future> getCurrentUser() async {
    try {
      final response = await _dio.get('/auth/user');

      if (response.statusCode == 200) {
        final data = response.data;
        if (data['success'] == true) {
          return ApiResponse.success(User.fromJson(data['data']['user']));
        }
      }

      return ApiResponse.error('Failed to get user data');
    } catch (e) {
      return ApiResponse.error('Network error: $e');
    }
  }
}

⚠️ Common Pitfalls to Avoid

❌ DON'T:

  • Store tokens in SharedPreferences
  • Hardcode API URLs
  • Ignore SSL certificates
  • No error handling
  • No token refresh logic

✅ DO:

  • Use FlutterSecureStorage for tokens
  • Environment-specific configurations
  • Implement retry logic
  • Handle offline scenarios
  • Use proper BLoC state management

🚨 Critical Integration Notes

  • Headers Required: Always include Authorization: Bearer YOUR_TOKEN and Accept: application/json
  • Token Format: Use flutter_secure_storage for token storage, never shared_preferences
  • Base URL: Use https://yourdomain.com/api for production, http://10.0.2.2:8000/api for Android emulator
  • Error Handling: All endpoints return JSON format, even errors: {"success": false, "message": "Error description"}

// lib/blocs/auth/auth_bloc.dart
class AuthBloc extends Bloc {
  final AuthRepository _authRepository;

  AuthBloc(this._authRepository) : super(AuthInitial()) {
    on(_onLoginRequested);
    on(_onUserRequested);
    on(_onLogoutRequested);
  }

  Future _onLoginRequested(
    AuthLoginRequested event,
    Emitter emit,
  ) async {
    emit(AuthLoading());

    final result = await _authRepository.login(event.login, event.password);

    if (result.isSuccess) {
      emit(AuthAuthenticated(result.data!));
    } else {
      emit(AuthError(result.error!));
    }
  }

  Future _onUserRequested(
    AuthUserRequested event,
    Emitter emit,
  ) async {
    emit(AuthLoading());

    final result = await _authRepository.getCurrentUser();

    if (result.isSuccess) {
      emit(AuthUserLoaded(result.data!));
    } else {
      emit(AuthError(result.error!));
    }
  }
}

// lib/blocs/chat/chat_bloc.dart
class ChatBloc extends Bloc {
  final ChatRepository _chatRepository;

  ChatBloc(this._chatRepository) : super(ChatInitial()) {
    on(_onChatListRequested);
    on(_onMessageSent);
  }

  Future _onMessageSent(
    MessageSent event,
    Emitter emit,
  ) async {
    final result = await _chatRepository.sendMessage(event.chatId, event.content);

    if (result.isSuccess) {
      // Add message to current chat state
      final currentState = state;
      if (currentState is ChatMessagesLoaded) {
        final updatedMessages = [result.data!, ...currentState.messages];
        emit(ChatMessagesLoaded(
          currentState.chat,
          updatedMessages,
          currentState.hasMore,
        ));
      }
    } else {
      emit(ChatError(result.error!));
    }
  }
}

// lib/screens/login_screen.dart
class LoginScreen extends StatefulWidget {
  @override
  _LoginScreenState createState() => _LoginScreenState();
}

class _LoginScreenState extends State {
  final _formKey = GlobalKey();
  String _login = '';
  String _password = '';
  bool _isLoading = false;

  Future _handleLogin() async {
    if (!_formKey.currentState!.validate()) return;

    setState(() => _isLoading = true);

    final authBloc = BlocProvider.of(context);
    authBloc.add(AuthLoginRequested(_login, _password));
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text('Login')),
      body: BlocListener(
        listener: (context, state) {
          if (state is AuthAuthenticated) {
            Navigator.pushReplacementNamed(context, '/home');
          } else if (state is AuthError) {
            ScaffoldMessenger.of(context).showSnackBar(
              SnackBar(content: Text(state.message)),
            );
          }
        },
        child: Form(
          key: _formKey,
          child: Column(
            children: [
              TextFormField(
                decoration: InputDecoration(labelText: 'Email or Phone'),
                onChanged: (value) => _login = value,
                validator: (value) => value!.isEmpty ? 'Required' : null,
              ),
              TextFormField(
                decoration: InputDecoration(labelText: 'Password'),
                obscureText: true,
                onChanged: (value) => _password = value,
                validator: (value) => value!.isEmpty ? 'Required' : null,
              ),
              BlocBuilder(
                builder: (context, state) {
                  if (state is AuthLoading) {
                    return CircularProgressIndicator();
                  }
                  return ElevatedButton(
                    onPressed: _handleLogin,
                    child: Text('Login'),
                  );
                },
              ),
            ],
          ),
        ),
      ),
    );
  }
}

// lib/services/websocket_service.dart
class WebSocketService {
  static const String _baseUrl = 'ws://yourdomain.com:8080';
  WebSocketChannel? _channel;
  final StreamController> _messageController =
    StreamController>.broadcast();

  Stream> get messages => _messageController.stream;

  Future connect(String token) async {
    try {
      _channel = WebSocketChannel.connect(
        Uri.parse('$_baseUrl?token=$token'),
      );

      _channel!.stream.listen(
        (data) {
          final message = json.decode(data);
          _messageController.add(message);
        },
        onError: (error) {
          print('WebSocket error: $error');
        },
        onDone: () {
          print('WebSocket connection closed');
        },
      );
    } catch (e) {
      print('Failed to connect to WebSocket: $e');
    }
  }

  void sendMessage(Map data) {
    if (_channel != null) {
      _channel!.sink.add(json.encode(data));
    }
  }

  void disconnect() {
    _channel?.sink.close();
    _messageController.close();
  }
}

      return ApiResponse.error('Network error: $e');
    }
  }

  // Upload avatar
  Future> uploadAvatar(File avatarFile) async {
    try {
      final formData = FormData.fromMap({
        'avatar': await MultipartFile.fromFile(avatarFile.path),
      });

      final response = await _dio.post('/media/upload/avatar', data: formData);

      if (response.statusCode == 200) {
        final data = response.data;
        if (data['success'] == true) {
          return ApiResponse.success(Media.fromJson(data['data']));
        }
      }

      return ApiResponse.error('Failed to upload avatar');
    } catch (e) {
      return ApiResponse.error('Network error: $e');
    }
  }

  // Delete media
  Future> deleteMedia(String mediaUrl) async {
    try {
      final response = await _dio.delete('/media/delete', data: {
        'media_url': mediaUrl
      });

      if (response.statusCode == 200) {
        final data = response.data;
        if (data['success'] == true) {
          return ApiResponse.success(true);
        }
      }

      return ApiResponse.error('Failed to delete media');
    } catch (e) {
      return ApiResponse.error('Network error: $e');
    }
  }
}

// lib/services/notification_service.dart
class NotificationService {
  static const AndroidNotificationChannel channel = AndroidNotificationChannel(
    'high_importance_channel',
    'High Importance Notifications',
    description: 'This channel is used for important notifications.',
    importance: Importance.high,
  );

  final FlutterLocalNotificationsPlugin flutterLocalNotificationsPlugin =
    FlutterLocalNotificationsPlugin();

  Future init() async {
    const AndroidInitializationSettings initializationSettingsAndroid =
      AndroidInitializationSettings('@mipmap/ic_launcher');

    const InitializationSettings initializationSettings =
      InitializationSettings(android: initializationSettingsAndroid);

    await flutterLocalNotificationsPlugin.initialize(initializationSettings);
  }

  Future showNotification({
    required int id,
    required String title,
    required String body,
  }) async {
    await flutterLocalNotificationsPlugin.show(
      id,
      title,
      body,
      NotificationDetails(
        android: AndroidNotificationDetails(
          channel.id,
          channel.name,
          channelDescription: channel.description,
          importance: Importance.high,
          priority: Priority.high,
        ),
      ),
    );
  }
}
