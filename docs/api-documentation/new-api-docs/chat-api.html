POST /chats

  "participants": [2, 3, 4],
  "type": "group",
  "name": "Project Team",
  "description": "Team discussion"

üíª Code Examples

JavaScript (Real-time Chat)

// Get user's chats
const getChats = async () => {
  const response = await fetch('http://127.0.0.1:8000/api/chats', {
    headers: { 'Authorization': `Bearer ${token}` }
  });
  return await response.json();
};

// Create group chat
const createGroup = async (participants, name) => {
  const response = await fetch('http://127.0.0.1:8000/api/chats', {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${token}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      participants,
      type: 'group',
      name
    })
  });
  return await response.json();
};

// Send message
const sendMessage = async (chatId, content) => {
  const response = await fetch(`http://127.0.0.1:8000/api/chats/${chatId}/messages`, {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${token}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      type: 'text',
      content
    })
  });
  return await response.json();
};

// Add reaction
const addReaction = async (chatId, messageId, emoji) => {
  const response = await fetch(`http://127.0.0.1:8000/api/chats/${chatId}/messages/${messageId}/react`, {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${token}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({ emoji })
  });
  return await response.json();
};

cURL Commands

# Get chat list
curl -X GET "http://127.0.0.1:8000/api/chats" \
  -H "Authorization: Bearer TOKEN"

# Create group chat
curl -X POST "http://127.0.0.1:8000/api/chats" \
  -H "Authorization: Bearer TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "participants": [2, 3],
    "type": "group",
    "name": "Project Team"
  }'

# Send message
curl -X POST "http://127.0.0.1:8000/api/chats/1/messages" \
  -H "Authorization: Bearer TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "type": "text",
    "content": "Hello everyone!"
  }'

# Add reaction
curl -X POST "http://127.0.0.1:8000/api/chats/1/messages/5/react" \
  -H "Authorization: Bearer TOKEN" \
  -H "Content-Type: application/json" \
  -d '{"emoji": "üëç"}'

# Pin chat
curl -X POST "http://127.0.0.1:8000/api/chats/1/pin" \
  -H "Authorization: Bearer TOKEN"
